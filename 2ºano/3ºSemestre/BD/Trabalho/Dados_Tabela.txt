create schema trabalho

create table trabalho.especie (
	nomee varchar(50) primary key,
	classe varchar(50)
);

create table trabalho.animal (
	nomea varchar(50) primary key,
	genero varchar (50),
	nomee varchar(50),
	foreign key (nomee) references trabalho.especie on delete restrict
);

create table trabalho.compartimento  (
	idcomp integer primary key,
	tipo varchar(50)
);


create table trabalho.tratador(
	ncc integer primary key,
	nomet varchar(50),
	salario integer,
	nccchefe integer
);
create table trabalho.alojado (
	nomea varchar(50),
	idcomp integer,
	primary key (nomea,idcomp),
	foreign key  (idcomp) references trabalho.compartimento on delete restrict
);
create table trabalho.trata(
	idcomp integer,
	ncc integer,
	primary key (idcomp,ncc),
	foreign key (idcomp) references trabalho.compartimento on delete restrict,
	foreign key (ncc) references trabalho.tratador on delete restrict
);
